11. WAP that takes two operands and one operator from the user and perform the operation and prints the result by using Switch statement

#include<stdio.h>

#include<conio.h>

int main()

{

int a,b;char ch;

printf("Enter '+'' for addition\n");

printf("Enter '-' for subtraction\n");

printf("Enter '*' for multiplication\n");

printf("Enter '/' for quotient\n");

printf("Enter '%%' for remainder\n");

printf("Enter any operator \n");

scanf("%c",&ch);

printf("Enter the value of any two operands such that (a>b)\n");

scanf("%d%d",&a,&b);

switch(ch)

{

case '+':

printf("Addition of %d and %d is=%d",a,b,(a+b));

break;

case '-':

printf("Subtraction of %d and %d is=%d",a,b,(a-b));

break;

case '*':

printf("Multiplication of %d and %d is=%d",a,b,(a*b));

break;

case '/':

printf("Quotient of %d and %d is=%d",a,b,(a/b));

break;

case '%':

printf("Remainder of %d and %d is=%d",a,b,(a%b));

break;

default:

printf("Wrong choice\n");

}

return 0;

getch();

}
