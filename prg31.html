31. WAP to check whether a given word exists in a file or not.If yes then find the number of times it occurs.

Remark : Create file of name " TEXT.txt " in the folder where you run this code.

#include<conio.h>

#include<stdio.h>

#define Buffer_Size 100

int main()

{

FILE *fp;

char st[Buffer_Size],s[50];int k=0;

printf("Enter the word to be searched\n");

scanf("%s",&s);

fp=fopen("TEXT.txt","r");

if(fp==NULL)

{

printf("File not found\n");

}

while(fgets(st,40,fp)!=NULL)

{

printf("%s",st);

if(strcmp(st,s)==1)

{

k++;

}

}

printf("\nThe number of times searched word present is=%d",k);

fclose(fp);

}
