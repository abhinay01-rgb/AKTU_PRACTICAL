32. WAP to compare the content of two files and determine whether they are same or not.

Remark : Create two files of name " TEXT.txt " and " TEXT1.txt " in the folder where you run this code.

#include<stdio.h>

#include<conio.h>

int main()

{

FILE * fPtr1;

FILE * fPtr2;

char ch1, ch2;

fPtr1 = fopen("TEXT.txt", "r");

fPtr2 = fopen("TEXT1.txt", "r");

if (fPtr1 == NULL || fPtr2 == NULL)

{

printf("\nUnable to open file.\n");

exit(EXIT_FAILURE);

}

do

{

ch1 = fgetc(fPtr1);

ch2 = fgetc(fPtr2);

} while (ch1 != EOF && ch2 != EOF);

if (ch1 == EOF && ch2 == EOF)

{

printf("\nBoth files are equal.");

}

else

{

printf("\nFiles are not equal.\n");

}

fclose(fPtr1);

fclose(fPtr2);

return 0;

}

getch();

}
